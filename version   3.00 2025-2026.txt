//--- Helper functions ---

double PipPoint() {
   if(_Digits == 3 || _Digits == 5)
      return (_Point * 10.0);
   return _Point;
}
double MinStopDistancePrice() {
   double stops  = (double)SymbolInfoInteger(_Symbol, SYMBOL_TRADE_STOPS_LEVEL) * _Point;
   double freeze = (double)SymbolInfoInteger(_Symbol, SYMBOL_TRADE_FREEZE_LEVEL) * _Point;
   return MathMax(stops, freeze) + (2.0 * _Point);
}
int VolumeDigits() {
   double step = SymbolInfoDouble(_Symbol, SYMBOL_VOLUME_STEP);
   if(step <= 0.0)
      return 2;
   int digits = 0;
   while(step < 1.0 && digits < 8) { step *= 10.0; digits++; }
   return digits;
}
double NormalizeVolume(double lots) {
   double minLot = SymbolInfoDouble(_Symbol, SYMBOL_VOLUME_MIN);
   double maxLot = SymbolInfoDouble(_Symbol, SYMBOL_VOLUME_MAX);
   double step   = SymbolInfoDouble(_Symbol, SYMBOL_VOLUME_STEP);
   if(step <= 0.0) step = 0.01;
   lots = MathMax(minLot, MathMin(maxLot, lots));
   lots = MathFloor(lots / step) * step;
   return NormalizeDouble(lots, VolumeDigits());
}
bool IsTradingSession() {
   if(!UseSessionFilter) return true;
   MqlDateTime tm; TimeToStruct(TimeCurrent(), tm);
   return (tm.hour >= LondonOpenHour && tm.hour < NYCloseHour);
}
int FindFreeTradeSlot() {
   for(int i=0; i<ArraySize(g_Trades); i++)
      if(g_Trades[i].ticket == 0) return i;
   return -1;
}
int CountActiveTrades() {
   int cnt = 0;
   for(int i=0; i<ArraySize(g_Trades); i++)
      if(g_Trades[i].ticket != 0 && PositionSelectByTicket(g_Trades[i].ticket))
         cnt++;
   return cnt;
}

void BuildEntryStops(ENUM_ORDER_TYPE type, double &sl, double &tp) {
   double pip = PipPoint(), minDist = MinStopDistancePrice();
   double bid = SymbolInfoDouble(_Symbol, SYMBOL_BID);
   double ask = SymbolInfoDouble(_Symbol, SYMBOL_ASK);
   double price = (type == ORDER_TYPE_BUY) ? ask : bid;
   double slDist = MathMax(StopLossPips * pip, minDist);
   double tpDist = MathMax(TakeProfitPips * pip, minDist);

   if(type == ORDER_TYPE_BUY) { sl = price - slDist; tp = price + tpDist; }
   else { sl = price + slDist; tp = price - tpDist; }
   sl = NormalizeDouble(sl, _Digits); tp = NormalizeDouble(tp, _Digits);
}

void ManagePositionProtection(ulong ticket) {
   if(!PositionSelectByTicket(ticket)) return;
   int type = (int)PositionGetInteger(POSITION_TYPE);
   double open = PositionGetDouble(POSITION_PRICE_OPEN);
   double curSL = PositionGetDouble(POSITION_SL);
   double pip = PipPoint();
   double bid = SymbolInfoDouble(_Symbol, SYMBOL_BID);
   double ask = SymbolInfoDouble(_Symbol, SYMBOL_ASK);
   double profitP = (type==POSITION_TYPE_BUY) ? (bid-open)/pip : (open-ask)/pip;

   // Break-even logic
   if(UseBreakEven && profitP >= BreakEvenTriggerPips) {
      double beSL = (type==POSITION_TYPE_BUY)
         ? open+BreakEvenPlusPips*pip
         : open-BreakEvenPlusPips*pip;
      if(curSL == 0.0 ||
         (type==POSITION_TYPE_BUY && curSL < beSL) ||
         (type==POSITION_TYPE_SELL && curSL > beSL))
         trade.PositionModify(ticket, beSL, PositionGetDouble(POSITION_TP));
   }

   // Trailing
   if(UseTrailingStop && profitP >= TrailingStartPips) {
      double trailSL = (type==POSITION_TYPE_BUY)
            ? bid - TrailingStepPips * pip
            : ask + TrailingStepPips * pip;
      if(curSL == 0.0 ||
         (type==POSITION_TYPE_BUY && curSL < trailSL) ||
         (type==POSITION_TYPE_SELL && curSL > trailSL))
         trade.PositionModify(ticket, trailSL, PositionGetDouble(POSITION_TP));
   }
}

void ManageTrades() {
   for(int i=0;i<ArraySize(g_Trades);i++) {
      ulong t = g_Trades[i].ticket;
      if(t == 0) continue;
      if(!PositionSelectByTicket(t)) { g_Trades[i].ticket = 0; continue; }
      ManagePositionProtection(t);
   }
}

//--- Signals and Trading ---
bool IsNewCandle() {
   datetime t = iTime(_Symbol, _Period, 0);
   if(t == lastCandleTime) return false;
   lastCandleTime = t;
   return true;
}

ENUM_TRADE_SIGNAL GetSignal() {
   if(CopyBuffer(FastMAHandle,0,0,3,FastMA)<3) return SIGNAL_NEUTRAL;
   if(CopyBuffer(SlowMAHandle,0,0,3,SlowMA)<3) return SIGNAL_NEUTRAL;
   if(FastMA[2]<=SlowMA[2] && FastMA[1]>SlowMA[1]) return SIGNAL_BUY;
   if(FastMA[2]>=SlowMA[2] && FastMA[1]<SlowMA[1]) return SIGNAL_SELL;
   return SIGNAL_NEUTRAL;
}

void OpenTradesForSignal(ENUM_TRADE_SIGNAL signal) {
   if(signal == SIGNAL_NEUTRAL) return;
   if(!IsTradingSession()) return;
   int active = CountActiveTrades();
   int slotsLeft = g_MaxPairsRuntime - active;
   if(slotsLeft<=0) return;
   int toOpen = MathMin(g_TradesPerSignalRuntime, slotsLeft);
   ENUM_ORDER_TYPE otype = (signal == SIGNAL_BUY) ? ORDER_TYPE_BUY : ORDER_TYPE_SELL;
   for(int n=0;n<toOpen;n++) {
      int idx = FindFreeTradeSlot();
      if(idx < 0) break;
      double sl=0, tp=0;
      BuildEntryStops(otype, sl, tp);
      bool ok=false;
      if(otype==ORDER_TYPE_BUY) ok=trade.Buy(LotSize,_Symbol,0,sl,tp);
      else ok=trade.Sell(LotSize,_Symbol,0,sl,tp);
      if(ok) g_Trades[idx].ticket = trade.ResultOrder();
      if(DebugMode)
        PrintFormat("[%s] Opened %s slot=%d lots=%.2f ticket=%I64u",
          EA_Name, EnumToString(otype), idx, LotSize, trade.ResultOrder());
   }
   // Optional: alert, notification, or email
   string msg = EA_Name + " " + _Symbol + " -> " + (signal==SIGNAL_BUY ? "BUY":"SELL") + " signal";
   if(EnableAlerts) Alert(msg);
   if(EnableNotifications) SendNotification(msg);
   if(EnableEmail) SendMail(EA_Name + " Signal", msg);
   Print(msg);
}

int OnInit() {
 // ...
   InitialBalance = AccountInfoDouble(ACCOUNT_BALANCE);
   MaxDrawdownHit = false;
   Print(EA_Name, ": Init v3.00");
   trade.SetExpertMagicNumber(MagicNumber);
   trade.SetDeviationInPoints(20);
   trade.SetTypeFillingBySymbol(_Symbol);
   FastMAHandle = iMA(_Symbol, _Period, MAFastPeriod, 0, MAFastMethod, MAFastPrice);
   SlowMAHandle = iMA(_Symbol, _Period, MASlowPeriod, 0, MASlowMethod, MASlowPrice);
   if(FastMAHandle == INVALID_HANDLE || SlowMAHandle == INVALID_HANDLE) {
      Print("MA handle error");
      return(INIT_FAILED);
        InitialBalance = AccountInfoDouble(ACCOUNT_BALANCE);
    ProfitTargetHit = false;
    MaxDrawdownHit = false;
   }
   ArraySetAsSeries(FastMA, true);
   ArraySetAsSeries(SlowMA, true);
   g_MaxPairsRuntime = MathMax(1, MaxPairs);
   g_TradesPerSignalRuntime = MathMax(1, TradesPerSignal);
   ArrayResize(g_Trades, g_MaxPairsRuntime);
   for(int i=0;i<ArraySize(g_Trades);i++) g_Trades[i].ticket=0;
   return(INIT_SUCCEEDED);
}


void OnDeinit(const int reason) {
   if(FastMAHandle != INVALID_HANDLE) IndicatorRelease(FastMAHandle);
   if(SlowMAHandle != INVALID_HANDLE) IndicatorRelease(SlowMAHandle);
}

void OnTick() {
    double equity = AccountInfoDouble(ACCOUNT_EQUITY);

    // ---- PROFIT TARGET ----
    if(ProfitTargetPercent > 0 && !ProfitTargetHit) {
        double profitPercent = 100.0 * (equity - InitialBalance) / InitialBalance;
        if(profitPercent >= ProfitTargetPercent) {
            ProfitTargetHit = true;
            string msg = StringFormat("[%s] PROFIT TARGET reached: %.2f%% (target: %.2f%%).", EA_Name, profitPercent, ProfitTargetPercent);
            Print(msg);
            if(EnableProfitTargetNotification) {
                SendNotification(msg); // Always notify
                SendMail(EA_Name + " Profit Target", msg);
                Alert(msg);
            }
            if(ProfitTargetAction == 1) {
                // Close all open trades, then stop further trading
                for(int i=0;i<ArraySize(g_Trades);i++) {
                    ulong t = g_Trades[i].ticket;
                    if(t != 0 && PositionSelectByTicket(t))
                        trade.PositionClose(t);
                }
                return;
            }
        }
    }
    if(ProfitTargetAction == 1 && ProfitTargetHit) return;

    // ---- MAX LOSS/DRAWDOWN ----
    if(MaxDrawdownPercent > 0 && !MaxDrawdownHit) {
        double percentDrawdown = 100.0 * (InitialBalance - equity) / InitialBalance;
        if(percentDrawdown >= MaxDrawdownPercent) {
            MaxDrawdownHit = true;
            string msg = StringFormat("[%s] MAX LOSS/DRAWDOWN triggered! Equity drawdown = %.2f%%, Max = %.2f%% -- Trading stopped.", EA_Name, percentDrawdown, MaxDrawdownPercent);
            Print(msg);
            if(EnableMaxLossNotification) {
                SendNotification(msg); // Always notify
                SendMail(EA_Name + " Max Loss", msg);
                Alert(msg);
            }
            // Close all open trades:
            for(int i=0;i<ArraySize(g_Trades);i++) {
                ulong t = g_Trades[i].ticket;
                if(t != 0 && PositionSelectByTicket(t))
                    trade.PositionClose(t);
            }
            return;
        }
    }
    if(MaxDrawdownHit) return;

    // ---- YOUR NORMAL TRADING LOGIC FOLLOWS ----
    ManageTrades();
    if(!IsNewCandle()) return;
    ENUM_TRADE_SIGNAL sig = GetSignal();
    OpenTradesForSignal(sig);
}
